ترتيب قيم الجدول

هي ترتيب قيم الجدول (تصاعديا) يعني من اصغر قيمة الى اكبر قيمة ,هناك عدة خوارزميات للترتيب الجدول سواء تنازليا او تصاعديا لكن لست مدطرا ان تفكر في الخوارزمية لان اغلب لغات البرمجة لديها دوال لترتيب قيم الجدول مثل (java, javascript,python ...etc)

//Code javaScript
tab = [1,0,9,-3,4,8,9,1,10,22,5];
tab = tab. sort( function (a, b) { return a - b });
console. log(tab)

Result

[-3, 0, 1, 1, 4, 5, 8, 9, 9, 10, 22]
//Code javaScript
tab = [1,0,9,-3,4,8,9,1,10,22,5];
tab = tab. sort( function (a, b) { return a - b });
//استخراج عدد قيم الجدول
var len = tab.length;
//استخراج اخر قيمة في الجدول
var max = tab[ len - 1 ];
//استخراج اول قيمة في الجدول
var min = tab[ 0 ];
console. log("max = "+max);
console. log("min = "+min);

Result

max = 22
min = -3

: (Mean) استخراج المعدل

اسنخراج المعدل هو ببساطة جمع كل قيم الجدول وتقسيمها على عدد قيم الجدول

//javaScript code
//دالة حساب المعدل
function Mean(tab) {
var somme = 0;
//جمع جميع قيم الجدول
for ( var i in tab) {
somme = somme + tab[i];
}
// تقسيم ناتج الجمع على عدد قيم الجدول
return (somme / tab.length).toFixed(2);
//toFixed
// ملاحظة : هي دالة للتحديد عدد الارقام بعد الفاصلة
}
tab = [1,0,9,-3,4,8,9,1,10,22,5];
console. log("mean = " + Mean(tab));

Result

mean = 6.00

: (Median (Q2)) استخراج الوسيط

بعد ترتيب الجدول لديك حالتين في حساب الوسيط :
الحالة الاولى هي عدد القيم فردي: اذا ببساطة القيمة التي في وسط الجدول هي الوسيط
الحالة الثانية هي عدد القيم زوجي: في هذه الحالة الوسيط هو معدل القيمتين اللتان في نصف الجدول اي نقسم القيمتين اللتان في نصف الجدول على اثنين
لكن كيف نعرف اذا كان العدد زوجي او فردي في البرمجة ?
عادتا ما يستعمل باقي القسمة على 2
اذا كان الباقي يساوي 0 فهو عدد زوجي
اذا كان الباقي يساوي 1 فهو عدد فردي
اذا اولا يجب معرفة اذا كان عدد القيم زوجي او فردي ثم نستخرج الوسيط .
